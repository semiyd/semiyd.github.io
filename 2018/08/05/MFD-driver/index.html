MFD driver


MFD子系统和MFD驱动介绍
`
MFD是multi-function device的简称。
有这么一类外设，其功能不止一种，内核把这类设备，统一称为mfd设备。
这类设备包括：
PMIC 通常内部包括regulator,rtc,gpio,wdg,motor,ON key等等子功能。
USB转i2c/spi/gio芯片
等等

向mfd子系统注册的api如下：
int mfd_add_devices(struct device parent, int id, const struct mfd_cell cells, int n_devs, struct resource mem_base, int irq_base, struct irq_domain irq_domain);
void mfd_remove_devices(struct device parent);
这其中有个很重要的结构体，是mfd_cell
struct mfd_cell {
const char name;
int id;

/* refcounting for multiple drivers to use a single cell */
atomic_t        *usage_count;
int            (*enable)(struct platform_device *dev);
int            (*disable)(struct platform_device *dev);

int            (*suspend)(struct platform_device *dev);
int            (*resume)(struct platform_device *dev);

/* platform data passed to the sub devices drivers */
void            *platform_data;
size_t            pdata_size;

/* device properties passed to the sub devices drivers */
struct property_entry *properties;

/*
 * Device Tree compatible string
 * See: Documentation/devicetree/usage-model.txt Chapter 2.2 for details
 */
const char        *of_compatible;

/* Matches ACPI */
const struct mfd_cell_acpi_match    *acpi_match;

/*
 * These resources can be specified relative to the parent device.
 * For accessing hardware you should use resources from the platform dev
 */
int            num_resources;
const struct resource    *resources;

/* don't check for resource conflicts */
bool            ignore_resource_conflicts;

/*
 * Disable runtime PM callbacks for this subdevice - see
 * pm_runtime_no_callbacks().
 */
bool            pm_runtime_no_callbacks;

/* A list of regulator supplies that should be mapped to the MFD
 * device rather than the child device when requested
 */
const char * const    *parent_supplies;
int            num_parent_supplies;

};
这里所谓cell，就是描述mdf设备的一个子模块。例如一个pmic里面，会有很多cell。

以drivers/mfd/tps6507x.c的代码为例：
首先可以看到一个简单的i2c驱动：
static const struct i2c_device_id tps6507x_i2c_id[] = {
{ “tps6507x”, 0 },
{ }
};
MODULE_DEVICE_TABLE(i2c, tps6507x_i2c_id);

#ifdef CONFIG_OF
static const struct of_device_id tps6507x_of_match[] = {
{.compatible = “ti,tps6507x”, },
{},
};
MODULE_DEVICE_TABLE(of, tps6507x_of_match);

#endif
static struct i2c_driver tps6507x_i2c_driver = {
.driver = {
.name = “tps6507x”,
.of_match_table =
of_match_ptr(tps6507x_of_match),
},
.probe = tps6507x_i2c_probe,
.remove = tps6507x_i2c_remove,
.id_table = tps6507x_i2c_id,
};
static int init tps6507x_i2c_init(void)
{
return i2c_add_driver(&tps6507x_i2c_driver);
}
/ init early so consumer devices can complete system boot /
subsys_initcall(tps6507x_i2c_init);
static void exit tps6507x_i2c_exit(void)
{
i2c_del_driver(&tps6507x_i2c_driver);
}
module_exit(tps6507x_i2c_exit);
然后在tps6507x_i2c_probe里面，就涉及了向mdf子系统注册的内容了：
首先声明，这个mfd设备里面有哪些cell：
static const struct mfd_cell tps6507x_devs[] = {
{
.name = “tps6507x-pmic”,
},
{
.name = “tps6507x-ts”,
},
};
上面这个例子说明有两个cell，一个是pmic，一个是ts
static int tps6507x_i2c_probe(struct i2c_client i2c,
const struct i2c_device_id id)
{
struct tps6507x_dev tps6507x;
tps6507x = devm_kzalloc(&i2c->dev, sizeof(struct tps6507x_dev),
GFP_KERNEL);
if (tps6507x == NULL)
return -ENOMEM;
i2c_set_clientdata(i2c, tps6507x);
tps6507x->dev = &i2c->dev;
tps6507x->i2c_client = i2c;
tps6507x->read_dev = tps6507x_i2c_read_device;
tps6507x->write_dev = tps6507x_i2c_write_device;
return mfd_add_devices(tps6507x->dev, -1, tps6507x_devs, ARRAY_SIZE(tps6507x_devs), NULL, 0, NULL);
}
仔细分析下此处mfd_add_devices的入参的选择。
struct device parent使用的是tps6507x->dev
这里的tps6507x，是个驱动私有结构体，
struct tps6507x_dev {
struct device dev;
struct i2c_client i2c_client;
int (read_dev)(struct tps6507x_dev tps6507x, char reg, int size,
void dest);
int (write_dev)(struct tps6507x_dev tps6507x, char reg, int size,
void src);
[…]
};
此处传给mfd有啥用？其实就是把i2c_client{}和i2c操作的接口read_dev/write_dev这类信息，将来可以传给具体的Cell的驱动使用。
具体实现的方法，应该是mfd_add_devices会根据mfd_cell创建platform_device子设备，并且把这些cell的platform_device的parent指向这个调用mfd_add_devices的mfd设备。
（所以注意，此时mfd的mfd_cell设备的platform_device，非dts创建，而是mfd_add_devices创建）
将来在cell驱动初始化的时候，取出私有结构体的方法如下：
static int tps6507x_ts_probe(struct platform_device pdev)
{
struct tps6507x_dev tps6507x_dev = dev_get_drvdata(pdev->dev.parent);
[…]
};
static int tps6507x_pmic_probe(struct platform_device pdev)
{
struct tps6507x_dev tps6507x_dev = dev_get_drvdata(pdev->dev.parent);
[…]
};
可以看到，很简单，去找.parent就可以了。mfd已帮你搞定一切。
int id使用的是-1，这个含义同platform_device里面的id，-1表示就一个设备。
const struct mfd_cell *cells当然是传入了tps6507x_devs{}这个cell结构体了

上面是个简单的例子。没有涉及struct resource mem_base, int irq_base, struct irq_domain irq_domain这三个入参。

以其中一个rtc cell为例，这里就涉及了resources
这里的resources，和platform_data里面的resources是一个东西。
rtc的resources如下：
static struct resource da9063_rtc_resources[] = {
{
.name = “ALARM”,
.start = DA9063_IRQ_ALARM,
.end = DA9063_IRQ_ALARM,
.flags = IORESOURCE_IRQ,
},
{
.name = “TICK”,
.start = DA9063_IRQ_TICK,
.end = DA9063_IRQ_TICK,
.flags = IORESOURCE_IRQ,
}
};
可以看到，上面的两个resource，都属于中断资源。分别定义了名字叫ALARM的中断号，和TICK的中断号。
这个resource ，作为mfd_cell 的成员，会被注册到mfd子系统中去。所以此处的resource，是每个cell独有的。
将来rtc cell驱动初始化的时候，就可以用了：
static int da9063_rtc_probe(struct platform_device *pdev)
{
[…]
irq_alarm = platform_get_irq_byname(pdev, “ALARM”);
ret = devm_request_threaded_irq(&pdev->dev, irq_alarm, NULL,
da9063_alarm_event,
IRQF_TRIGGER_LOW | IRQF_ONESHOT,
“ALARM”, rtc);
if (ret) {
dev_err(&pdev->dev, “Failed to request ALARM IRQ %d: %d\n”, irq_alarm, ret);
return ret;
}
[…]
};
可以看到，cell驱动中，只需用platform_get_irq或者platform_get_irq_byname，就可获得irq资源也就是irq号
这里获取mfd设备的私有数据，甚至没有必要使用dev_get_drvdata(pdev->dev.parent);
mfd帮你把cell的resource事先设置好。等cell驱动probe的时候，直接用就可以了。

在dts里面，又如何体现mfd设备和其下面的各个子cell设备的关系呢？
其实可以通过父节点-子节点的关系体现出来，如下：
pmic@58 {
compatible = “dlg,da9063”; 此为mfd父设备
reg = <0x58>;
interrupt-parent = <&irqc0>;
interrupts = <2 irq_type_level_low="">;
interrupt-controller;

    rtc {
        compatible = "dlg,da9063-rtc";    此为mfd的cell子设备
    };

    wdt {
        compatible = "dlg,da9063-watchdog";    此为mfd的cell子设备
    };
};

说到这，基本清楚了，就是mfd设备，因为自带很多cell子设备。所以mfd设备会把共性和资源类的信息提炼出来。
将来cell子设备注册的时候，这些信息/函数等东西，早就被mfd父设备准备好了。这样代码重用性也比较好，也便于管理。
