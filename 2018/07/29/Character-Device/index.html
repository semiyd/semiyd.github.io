#define MKDEV(ma,mi) 生成一个dev节点，参数分别是major和minor值

register_chrdev_region(dev_t first,unsigned int count,char *name) 静态注册字符设备号
first :要分配的设备编号范围的初始值(次设备号常设为0); first由MKDEV宏生成。

*dev: 返回分配到的设备号;
firstminor:申请的次设备号的起始值
count：申请的次设备号的数目
name:编号相关联的设备名称. (/proc/devices);

struct cdev *cdev_alloc(void) 分配一个cdev结构体

void cdev_init(struct cdev cdev, struct file_operations fops); 初始化cdev结构体
cdev: cdev结构体指针
fops: file_operations指针

int cdev_add(struct cdev *dev, dev_t num, unsigned int count); 注册cdev字符设备

num: MKDEV或者alloc_chrdev_region生成的设备号
count： 通常是1。如果比1大，则意味着对应多个次设备号，不如磁带机，这不常见。

void unregister_chrdev_region(dev_t from, unsigned count) 注销字符设备号
from： 要注销的设备号;
count： 连续编号范围，一般为1
void cdev_del(struct cdev *dev); 注销cdev字符设备
dev： 要注销的cdev的指针

初始化的流程详解：
1.分配字符设备的设备号，alloc_chrdev_region
执行完就得到dev_t了。这步做了以后就不用MKDEV了。
如果不是第一次进来初始化字符设备，则不用alloc_chrdev_region，用register_chrdev_region注册minor即可，major使用alloc_chrdev_region申请到的。
2.MKDEV/MAJOR/MINOR宏，都是针对dev_t的移位操作，低20位是minor号
如果自己MKDEV，要保证major号至少是alloc_chrdev_region分配过的
3.alloc_chrdev_region做完后，cdev_init和cdev_add这两部做完。但/dev下还不会有设备。
此时sys下已有：/sys/devices/platform/soc/soc:xxx，假设dts中这个字符设别是放在soc的大节点内部的。
4.class_create：
生成/sys/class/xxx
5.device_create：（入参依赖第4步的class_create注册的class）
生成/dev/xxx
6.注意：alloc_chrdev_region/class_create/device_create/platform.driver.name保持一直，所以一般驱动会定义一个字符串的宏#define DEVNAME “xxx”
7.有关cdev_device_add的使用
cdev_device_add可以代替cdev_add+device_create的组合。
注意：在platform驱动的probe注册字符设备时，platform_device->device和字符设备的device结构体并无关系。
所以想用cdev_device_add，要先静态声明一个专属此字符设备的struct device，作为函数的第二个入参（可搜下内核相关代码）
做device_initialize，填充struct device（parent等），做device_set_name，最后再做cdev_device_add。
注意不能把platform_device->dev作为cdev_device_add的第二个入参！
总之cdev_device_add的使用和cdev_add+device_create基本相同。 90 ·

#define GLOBALMEM_SIZE 0x1000

#define GLOBALMEM_MAGIC ‘g’

#define MEM_CLEAR _IO(GLOBALMEM_MAGIC, 0)

#define GLOBALMEM_MAJOR 230

static int globalmem_major = GLOBALMEM_MAJOR;
module_param(globalmem_major, int, S_IRUGO);

struct globalmem_dev {
struct cdev cdev;
unsigned char mem[GLOBALMEM_SIZE];
};

struct globalmem_dev *globalmem_devp;

static int globalmem_open(struct inode inode, struct file filp)
{
filp->private_data = globalmem_devp;
return 0;
}

static int globalmem_release(struct inode inode, struct file filp)
{
return 0;
}

static long globalmem_ioctl(struct file filp, unsigned int cmd, unsigned long arg)
{
struct globalmem_dev dev = filp->private_data;

switch(cmd) {
    case MEM_CLEAR:
        memset(dev->mem, 0, GLOBALMEM_SIZE);
        break;
    default:
        return -EINVAL;
}

return 0;

}

static loff_t globalmem_llseek(struct file *filp, loff_t offset, int orig)
{
loff_t ret = 0;
switch(orig)
{
case 0://seek start from begining
if(offset < 0) {
ret = -EINVAL;
break;
}
if((unsigned int)offset > GLOBALMEM_SIZE) {
ret = -EINVAL;
break;
}
filp->f_pos = (unsigned int)offset;
ret = filp->f_pos;
break;
case 1://seek start from current position
if((filp->f_pos + offset) > GLOBALMEM_SIZE) {
ret = -EINVAL;
break;
}
if((filp->f_pos + offset) < 0) {
ret = -EINVAL;
break;
}
filp->f_pos += offset;
ret = filp->f_pos;
break;
case 2:
printk(“SEEK_END not supported!\n”);
ret = -EINVAL;
break;
default:
ret = -EINVAL;
break;
}
return ret;
}

static ssize_t globalmem_write(struct file filp, const char __user buf, size_t size, loff_t ppos)
{
unsigned long p = ppos;
unsigned int count = size;
int ret = 0;
struct globalmem_dev *dev = filp->private_data;

if(p >= GLOBALMEM_SIZE)
    return 0;
if(count > GLOBALMEM_SIZE - p)
    count = GLOBALMEM_SIZE - p;
if(copy_from_user(dev->mem + p, buf, count))
    ret = -EFAULT;
else {
    *ppos += count;
    ret = count;
    printk("write %u bytes from %lu\n",count,p);
}

return ret;

}

static ssize_t globalmem_read(struct file filp,char __user buf, size_t size, loff_t ppos)
{
unsigned long p = ppos;
unsigned int count = size;
int ret = 0;
struct globalmem_dev *dev = filp->private_data;

if(p >= GLOBALMEM_SIZE)
    return 0;
if(count > GLOBALMEM_SIZE - p)
    count = GLOBALMEM_SIZE - p;
if(copy_to_user(buf, dev->mem+p, count))
    {
        return -EFAULT;
    }
else
    {
        *ppos += count;
        ret = count;
        printk("read %u bytes from %lu\n",count,p);
    }
return ret;

}

static const struct file_operations globalmem_fops = {
.owner = THIS_MODULE,
.llseek = globalmem_llseek,
.read = globalmem_read,
.write = globalmem_write,
.unlocked_ioctl = globalmem_ioctl,
.open = globalmem_open,
.release = globalmem_release,
};

static void globalmem_setup_cdev(struct globalmem_dev *dev, int index)
{
int err, devno = MKDEV(globalmem_major, index);
cdev_init(&dev->cdev,&globalmem_fops);
dev->cdev.owner = THIS_MODULE;
err = cdev_add(&dev->cdev, devno, 1);
if(err)
{
printk(“error adding cdev!\n”);
}
}

static int __init globalmem_init(void)
{
int ret;
dev_t devno = MKDEV(globalmem_major, 0);

printk("globalmem_major = %d\n",globalmem_major);
if(globalmem_major)
    ret = register_chrdev_region(devno, 1, "globalmem");
else
    {
        ret = alloc_chrdev_region(&devno, 0, 1, "globalmem");
        globalmem_major = MAJOR(devno);
    }
if(ret < 0)
    {
        return ret;
    }
globalmem_devp = kzalloc(sizeof(struct globalmem_dev), GFP_KERNEL);
if(!globalmem_devp)
    {
        ret = -ENOMEM;
        goto fail_malloc;
    }

globalmem_setup_cdev(globalmem_devp, 0);
return 0;

fail_malloc:
unregister_chrdev_region(devno, 1);
return ret;

}

static void __exit globalmem_exit(void)
{
cdev_del(&globalmem_devp->cdev);
kfree(globalmem_devp);
unregister_chrdev_region(MKDEV(globalmem_major, 0), 1);
}

late_initcall(globalmem_init);
module_exit(globalmem_exit);

cat和read的异同点（记入kernel笔记）
cat: 会一直read，直到return 0为止。所以cat的本意是，读一个file，一直读，读到没有东西可以读为止。也就是一下子全部读出来。

两个常见的问题。
1.file_operations的open调用的时候，我想访问platform_device的私有设备结构体怎么办？
私有设备结构体xxx里面应该定义struct cdev成员。
cdev_init的时候，就是用这个cdev成员做初始化。
在open的时候，可以通过container_of，从inode->i_cdev的指针，去找到其上一级结构体指针xxx
类似这个例子：
static int mmc_rpmb_chrdev_open(struct inode inode, struct file filp)
{
struct mmc_rpmb_data *rpmb = container_of(inode->i_cdev,
struct mmc_rpmb_data, chrdev);

get_device(&rpmb->dev);
filp->private_data = rpmb;
mmc_blk_get(rpmb->md->disk);

return nonseekable_open(inode, filp);

}
2.ioctl的时候，怎么找到device结构体？
首先记得在platform_driver probe的时候，在私有设备结构体xxx里面保存下platform_device的指针
然后open的时候，把私有设备结构体xxx保存到file->private_data。
在ioctl的时候，就可以从file->private_data拿到私有设备结构体xxx，从而进一步拿到xxx->pdev->dev也就是device结构体
这点在ioctl的时候需要做runtime_pm操作，需要device结构体做入参的时候有用。
