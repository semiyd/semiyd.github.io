4.8.6 qt库 编译命令

Posted on 2018-07-25 | In Linux , Application , Qtopia , 4.8.6

1.
source environment-setup-cortexa7hf-neon-poky-linux-gnueabi

2.
change mkspecs/qws/linux-arm-g++/qmake.conf to:
#

qmake configuration for building with arm-linux-g++

#

include(../../common/linux.conf)
include(../../common/gcc-base-unix.conf)
include(../../common/g++-unix.conf)
include(../../common/qws.conf)

modifications to g++.conf

QMAKE_CC = arm-poky-linux-gnueabi-gcc
QMAKE_CXX = arm-poky-linux-gnueabi-g++
QMAKE_LINK = arm-poky-linux-gnueabi-g++
QMAKE_LINK_SHLIB = arm-poky-linux-gnueabi-g++

modifications to linux.conf

QMAKE_AR = arm-poky-linux-gnueabi-ar cqs
QMAKE_OBJCOPY = arm-poky-linux-gnueabi-objcopy
QMAKE_STRIP = arm-poky-linux-gnueabi-strip

QMAKE_INCDIR += /home/share/tslib/1.16/Board/arm-poky-linux-gnueabi-gcc-5.3.0-MYD-Y6ULX/usr/local/include/
QMAKE_LIBDIR += /home/share/tslib/1.16/Board/arm-poky-linux-gnueabi-gcc-5.3.0-MYD-Y6ULX/usr/local/lib/

QMAKE_CFLAGS += -mfloat-abi=hard
QMAKE_CXXFLAGS += -mfloat-abi=hard

load(qt_config)

3.
./configure -release -shared -fast -no-largefile -qt-sql-sqlite -no-qt3support -no-xmlpatterns -no-mmx -no-3dnow -no-sse -no-sse2 -no-svg -no-webkit -qt-zlib -qt-libtiff -qt-libpng -qt-libmng -qt-libjpeg -make libs -nomake tools -no-nis -no-cups -no-iconv -no-dbus -no-openssl -no-icu -xplatform qws/linux-arm-g++ -embedded arm -little-endian -qt-freetype -depths 16,18,24,32 -no-glib -qt-mouse-tslib -I /home/share/1.16/tslib/Board/arm-poky-linux-gnueabi-gcc-5.3.0-MYD-Y6ULX/usr/local/include/ -L /home/share/1.16/tslib/Board/arm-poky-linux-gnueabi-gcc-5.3.0-MYD-Y6ULX/usr/local/lib/ -prefix /home/share/Qt/4.8.6/Board/arm-poky-linux-gnueabi-gcc-5.3.0-MYD-Y6ULX/ -sysroot /home/share/cross-compiler/arm-poky-linux-gnueabi-gcc-5.3.0-MYD-Y6ULX/sysroots/cortexa7hf-neon-poky-linux-gnueabi -v

4.
make -j8
make install
