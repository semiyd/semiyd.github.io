main函数入参的处理



如何给一个应用加执行的参数？
比如
./abc –verbose

主要是依靠getopt_long来获取和解析入参。
举例如下：

#include

#include <time.h>

#include <string.h>

#include <stdio.h>

#include “System.h”

#include “version.h”

#include <getopt.h>

static void print_usage(void)
{
(void)fprintf(stderr,
“Usage: NetController \n”);
(void)fprintf(stderr,
“\t\t[ -v ] for verbose output on ring management library\n”);
}

#define SHORTOPTS “hv”

static const struct option longopts[] = {
{ “help”, no_argument, NULL, ‘h’ },
{ “verbose”, no_argument, NULL, ‘v’ },
{ NULL, 0, NULL, 0 }
};

int main(int argc, char* argv) {
time_t rawtime;
struct tm timeinfo;
char buffer[30];
time (&rawtime);
timeinfo = localtime(&rawtime);
strftime(buffer, 30, “%F %T”,timeinfo);
register int op;

while (
    (op = getopt_long(argc, argv, SHORTOPTS, longopts, NULL)) != -1)
    switch (op) {
        case 'v':
            ring_mgmt_debug = true;
            break;

        default:
            print_usage();
            return 0;
    }

//开始正式干活

}
