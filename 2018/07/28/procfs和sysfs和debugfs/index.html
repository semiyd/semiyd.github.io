        随后有关获取用户空间的数据，有两种方法：通过kfifo或者直接使用.
        总的来说，DEVICE_ATTR在定义属性的读写API时，其实传进来的时候，就已经是const char* buf了（参见下面的gt9xx_update_x的定义的入参）
        也就是用户空间输入的数字，到内核里面，已经是字符数组了。所以。如果要直接当成数字使用，是需要转换格式的。
        kfifo：
            #define PARROT_MSG_FIFO_SIZE 1024

            /* Use a Kernel FIFO for read operations */
            static DECLARE_KFIFO(gt9xx_msg_fifo_x, char, PARROT_MSG_FIFO_SIZE);
            static DECLARE_KFIFO(gt9xx_msg_fifo_y, char, PARROT_MSG_FIFO_SIZE);

            //记得在上面初始化的时候，同时初始化了这些fifo结构体
                 INIT_KFIFO(gt9xx_msg_fifo_x);
                 INIT_KFIFO(gt9xx_msg_fifo_y);

            /* Placing data into the read FIFO is done through sysfs */
            static ssize_t gt9xx_update_y(struct device* dev, struct device_attribute* attr, const char* buf, size_t count)
            {
             unsigned int copied;

             if (kfifo_avail(&gt9xx_msg_fifo_y) < count) {
              printk("not enough space left on fifo\n");
              return -ENOSPC;
             }

             /* The buffer is already in kernel space, so no need for ..._from_user() */
             copied = kfifo_in(&gt9xx_msg_fifo_y, buf, count);

             if (copied != count) {
              printk("short write detected\n");
             }

             return copied;
            }

            /* This sysfs entry resets the FIFO */
            static ssize_t gt9xx_simulate(struct device* dev, struct device_attribute* attr, const char* buf, size_t count)
            {
                char x;
                char y;
                unsigned int tmp=88;

                //kfifo_put(&gt9xx_msg_fifo_x, &tmp);
                //导出出来
                kfifo_out(&gt9xx_msg_fifo_x, &x, 3);
                kfifo_out(&gt9xx_msg_fifo_y, &y, 3);
                //清零
                kfifo_reset(&gt9xx_msg_fifo_x);
                kfifo_reset(&gt9xx_msg_fifo_y);
                 return count;
            }
            //字符设备的read函数示例，可以直接把刚刚收到的字符，返回到用户空间
            static ssize_t gt9xx_device_read(struct file* filp, char __user *buffer, size_t length, loff_t* offset)
            {
             int retval;
             unsigned int copied;

             if (kfifo_is_empty(&parrot_msg_fifo)) {
              return 0;
             }

             retval = kfifo_to_user(&parrot_msg_fifo, buffer, parrot_msg_len[parrot_msg_idx_rd], &copied);

             return retval ? retval : copied;
            }
         直接用strict_strtoul转换
            unsigned int x_axis = 0;
            static ssize_t gt9xx_update_x(struct device* dev, struct device_attribute* attr, const char* buf, size_t count)
            {
                if (strict_strtoul(buf, 10, &x_axis))
                    return -EINVAL;
                return count;
            }
            这样，在用户空间输入的数字，就直接被转成可以用的数字了，保存在x_axis。

bus-device-driver 模型：
    bus是指主控，如i2c主控制器
    device是指连接上来的外部设备
    driver是指驱动device的方法
